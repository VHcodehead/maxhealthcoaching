**3. Create `src/app/api/coach/route.ts` — lightweight coach lookup endpoint**

Returns the coach's userId and fullName for client-side use. Single-coach app, so find the first profile with role='coach':
```typescript
import { NextResponse } from 'next/server';
import { auth } from '@/lib/auth';
import { prisma } from '@/lib/db';

export async function GET() {
  const session = await auth();
  if (!session?.user) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });

  const coach = await prisma.profile.findFirst({
    where: { role: 'coach' },
    select: { userId: true, fullName: true }
  });
  if (!coach) return NextResponse.json({ error: 'No coach found' }, { status: 404 });
  return NextResponse.json(coach);
}
```
